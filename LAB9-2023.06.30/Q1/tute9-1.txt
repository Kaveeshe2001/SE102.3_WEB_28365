<head>
  <title>Display Day</title>
  <script>
    function getDayName(dayNum) {
    switch (dayNum) {
        case 1:
            return "Monday";
        case 2:
            return "Tuesday";
        case 3:
            return "Wednesday";
        case 4:
            return "Thursday";
        case 5:
            return "Friday";
        case 6:
            return "Saturday";
        case 7:
            return "Sunday";
        default:
            return "Invalid input! Please enter a number between 1 and 7.";
    }
    }

    function displayDay() {
    var num = window.prompt("Enter the Number from 1-7 :");
    var dayNum = parseInt(num);
    var dayName = getDayName(dayNum);
    window.alert(dayName);
    }
    displayDay();
  </script>
